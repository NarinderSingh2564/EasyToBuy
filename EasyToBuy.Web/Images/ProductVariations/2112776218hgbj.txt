CREATE VIEW [EasyToBuyAdmin].[view_GetProductList]
AS
SELECT        TP.Id, TP.ProductName, TP.ProductImage, TP.ProductDescription, TP.CategoryId, TP.TotalVolume, TC.PackingModeId, TPPM.PackingMode, TP.IsActive, TP.VendorId
FROM            dbo.tblProduct AS TP INNER JOIN
                         dbo.tblCategory AS TC ON TC.Id = TP.CategoryId INNER JOIN
                         dbo.tblProductPackingMode AS TPPM ON TPPM.Id = TC.PackingModeId

CREATE procedure [dbo].[spGetCartDetailsByCustomerId] @CustomerId int

as 

begin
        select    
		          TC.Id,
				  TPV.Id as VariationId,
				  TP.ProductName,
				  TV.Name as VendorName,
				  TP.ProductDescription,
			      TP.ProductImage,
				  TW.ProductWeight,
			      TC.Quantity,     
	              TC.Quantity * TPV.MRP as MRP,
				  TPV.Discount,
				  TC.Quantity * TPV.DiscountPrice as DiscountPrice,
				  TC.Quantity * TPV.PriceAfterDiscount as PriceAfterDiscount,
				  TPV.PriceAfterDiscount *  TC.Quantity as TotalProductPrice,
			      TPV.ShowProductWeight,
				  TPV.SetAsDefault

	              from tblCart TC
		       
				   inner join tblProductVariationAndRate TPV on TPV.Id = TC.VariationId
		           inner join tblUser TU on TU.Id = TC.UserId
			       inner join tblProduct TP on TP.Id = TPV.ProductId
				   inner join  tblVendor TV on TV.Id = TP.VendorId
				   inner join tblProductWeight TW on TW.Id = TPV.ProductWeightId

				 where TC.UserId = @CustomerId  and TC.IsPlaced = 0
end

CREATE procedure [dbo].[spGetOrderList] @CustomerId int ,@VendorId int ,  @SearchText varchar(15) , @StatusId int=null,@StartDate date ,@EndDate date 

as

begin

declare @queryText varchar(Max);

set @queryText=
				'select 
						tco.Id,
						tco.UserId,
				        tu.Name as UserName,
						tv.Id as VendorId,
				        tv.Name as VendorName,
						tco.OrderNumber,
						tco.OrderDate, 
						tp.Id as ProductId,
						tpv.Id as VariationId,
						tp.ProductName,
						tp.ProductImage,
						tpw.ProductWeight,
						tpp.PackingType,
						tco.Quantity,  
						tco.MRP,
						tco.Discount,
						tco.AmountToBePaid,
						tco.StatusId,
						tos.Status

				from tblCustomerOrder as tco
				inner join tblOrderStatus as tos on tos.Id = tco.StatusId
				inner join tblUser tu on tu.Id = tco.UserId
				inner join tblProductVariationAndRate tpv on tpv.Id = tco.VariationId
				inner join tblProduct as tp on tp.Id = tpv.ProductId
				inner join tblVendor tv on tv.Id = tp.VendorId
				inner join tblProductWeight tpw on tpw.Id = tpv.ProductWeightId
				inner join tblProductPacking tpp on tpp.Id = tpv.ProductPackingId
				where 1=1'  

				If(@CustomerId > 0)
				Begin
					set	@queryText = @queryText +' and tco.UserId = '+ cast(@CustomerId as varchar(10))
				End

					If(@VendorId > 0)
				Begin
					set	@queryText = @queryText +' and tp.VendorId = '+ cast(@VendorId as varchar(10))
				End
				
				If (@SearchText is not null)
			    Begin
				    set	@queryText = @queryText +' and (tp.ProductName like ''%'+@SearchText+'%'')'
				End

				If(@StatusId is not null)
				Begin
						If(@StatusId =5 OR @StatusId = 6)
						Begin
							set @queryText = @queryText + ' and tco.StatusId in('+(cast(@StatusId as varchar(10)))+')'
						End

						If(@StatusId <> 5 and @StatusId <> 6)
						Begin
							set @queryText = @queryText + ' and tco.StatusId not in(5,6)'
						End
				End

				If(@StartDate is not null and @EndDate is not null)
				Begin
					set @queryText = @queryText + ' and tco.OrderDate between ''' + (cast(@StartDate as varchar(50)) + ''' AND ''' + cast(@EndDate as varchar(50))) + ''''
				End 
					
				exec(@queryText)
		
					
				
end

CREATE procedure [dbo].[spGetProductList] @CategoryId int , @SearchText varchar(15),@VendorId  int null,@Role varchar(10) null
  
as 

begin

declare @queryText varchar(Max);

            If (@Role = 'Vendor')
			    Begin

				set @queryText = 'select    
		                                 TP.*,
		                                 TV.Name as VendorName,
	                                     cast(0 as decimal(18,2)) as MRP,
		                                 cast(0 as decimal(18,2)) as Discount,
		                                 cast(0 as decimal(18,2)) as DiscountPrice,
		                                 cast(0 as decimal(18,2)) as PriceAfterDiscount,
		                                 0 as VariationId,
		                                 cast(0 as varchar(10)) as ProductWeight,
		                                 cast(0 as bit) as ShowProductWeight,
		                                 1 as Quantity

	                               from view_GetProductList TP		
			                       inner join tblVendor TV on TV.Id = TP.VendorId where 1=1 
			                       and TP.VendorId = '+ cast(@VendorId as varchar(10));
				End

			Else
				Begin 

				set @queryText = 'select    
		                                 TP.*,
										 TV.Name as VendorName,
		                                 TPV.MRP,
		                                 TPV.Discount,
		                                 TPV.DiscountPrice,
		                                 TPV.PriceAfterDiscount,
		                                 TPV.Id as VariationId,
		                                 PW.ProductWeight,
		                                 TPV.ShowProductWeight,
		                                 1 as Quantity
	    
	                               from view_GetProductList TP		    
	                               inner join tblProductVariationAndRate TPV on TPV.ProductId = TP.Id
	                               inner join tblProductWeight PW on PW.Id = TPV.ProductWeightId
	                               inner join tblVendor TV on TV.Id = TP.VendorId where 1=1 and TP.IsActive = 1 and TPV.SetAsDefault = 1';
				End

				If(@CategoryId > 0)
				Begin
					set	@queryText = @queryText +' and TP.CategoryId = '+ cast(@CategoryId as varchar(10))
				End

			    If (@SearchText is not null)
			    Begin
				    set	@queryText = @queryText +' and (TP.ProductName like ''%'+@SearchText+'%'' or TC.CategoryName  like ''%'+@SearchText+'%'')'
				End
				exec(@queryText)
			
end

-- =============================================

-- Description:	<This sp is using to get Tracking Status List of any Order>
-- =============================================
CREATE PROCEDURE [dbo].[spGetTrackingStatusListByOrderId] @OrderId int
AS
BEGIN

	
select 
	       OS.Id, OS.Status , 
		   CASE
				WHEN COSL.id is not null THEN cast(1 as bit )
		        ELSE cast(0 as bit )
		   END AS IsPending,
		   Convert(varchar(30),COSL.CreatedOn,109) as StatusDate

		   from tblOrderStatus OS
		   Left Join tblcustomerorderstatuslog COSL on OS.id = COSL.statusid and COSL.orderid = @OrderId
		   order by OS.StatusNo asc
   
   
END

-- =============================================
-- Author:		<Arun Kumar>
-- Create date: <14/05/2024>
-- Description:	<Get Vendor Order Count like(Total Order, Cancel Order, Pending Order, Delivered Order)>
-- =============================================
CREATE procedure [dbo].[spGetVendorOrdersCountById]  @VendorId int
  
as 

begin

declare @VendorOrderCount int = 0

select @VendorOrderCount = count(TCO.Id) 
        from tblCustomerOrder TCO
       	inner join tblProductVariationAndRate TPV on TPV.Id = TCO.VariationId
        inner join tblProduct TP on TP.Id = TPV.ProductId
		inner join tblVendor TV on TV.Id = TP.VendorId
		group by TV.Id having(TV.Id = @VendorId)

if(@VendorOrderCount > 0)
begin
	Select  
	        1 as Id,TV.Id as VendorId,0 as OrderStatusId,'Total Orders' as OrderStatus,SUM(1) AS OrderCount,'bi bi-archive-fill fs-1 primary-text  rounded-full secondary-bg p-3 mycol' as IconClass
			
			from tblCustomerOrder TCO
	      	inner join tblProductVariationAndRate TPV on TPV.Id = TCO.VariationId
		    inner join tblProduct TP on TP.Id = TPV.ProductId
			inner join tblVendor TV on TV.Id = TP.VendorId
	        inner join tblOrderStatus TOS on TOS.Id  = TCO.StatusId
		    group by TV.Id having(TV.Id =  @VendorId)
	union all 
	Select     
			2 as Id,TV.Id as VendorId, 1 as OrderStatusId,'Pending' as OrderStatus,
			SUM(CASE WHEN TOS.Id = '1' or  TOS.Id = '2' or TOS.Id = '3' or TOS.Id = '4'  THEN 1 ELSE 0 END) AS OrderCount,'bi bi-hourglass fs-1 primary-text  rounded-full secondary-bg p-3' as IconClass
	        
			from tblCustomerOrder TCO
	     	inner join tblProductVariationAndRate TPV on TPV.Id = TCO.VariationId
			inner join tblProduct TP on TP.Id = TPV.ProductId
			inner join tblVendor TV on TV.Id = TP.VendorId
	        inner join tblOrderStatus TOS on TOS.Id  = TCO.StatusId
		    group by TV.Id having(TV.Id =  @VendorId ) 
	union all 
	Select     
			3 as Id,TV.Id as VendorId,5 as OrderStatusId,'Delivered' as OrderStatus,SUM(CASE WHEN TOS.Id = '5' THEN 1 ELSE 0 END) AS OrderCount,'bi bi-check-circle-fill fs-1 primary-text  rounded-full secondary-bg p-3' as IconClass
		    
			from tblCustomerOrder TCO
	        inner join tblProductVariationAndRate TPV on TPV.Id = TCO.VariationId
			inner join tblProduct TP on TP.Id = TPV.ProductId
			inner join tblVendor TV on TV.Id = TP.VendorId
	        inner join tblOrderStatus TOS on TOS.Id  = TCO.StatusId
		    group by TV.Id having(TV.Id =  @VendorId  )	
	union all 
	Select     
			4 as Id,TV.Id as VendorId,6 as OrderStatusId,'Cancelled ' as OrderStatus,SUM(CASE WHEN TOS.Id = '6' THEN 1 ELSE 0 END) AS OrderCount,'bi bi-slash-circle fs-1 primary-text  rounded-full secondary-bg p-3' as IconClass
	        
			from tblCustomerOrder TCO
	     	inner join tblProductVariationAndRate TPV on TPV.Id = TCO.VariationId
            inner join tblProduct TP on TP.Id = TPV.ProductId
		    inner join tblVendor TV on TV.Id = TP.VendorId
	        inner join tblOrderStatus TOS on TOS.Id  = TCO.StatusId
		    group by TV.Id having(TV.Id =  @VendorId  )
end

else
begin
	Select  
	        1 as Id, @VendorId  as VendorId,0 as OrderStatusId,'Total Orders' as OrderStatus,0 AS OrderCount,'bi bi-archive-fill fs-1 primary-text  rounded-full secondary-bg p-3 mycol' as IconClass
	union all
